36/// Macro that AETimer uses for the tick rounding functionality
37/// @attention This is an internal macro intended for internal AETimer's use.
38/// Isn't recommended to mess around with it
39/// @see #AETM_TICK_ROUNDING_METHOD
40#if AETM_TICK_ROUNDING_METHOD == 0 // round down/truncate
41 #define _AETM_TICK_ROUND(tick) ace::math::truncToInt<ullint>(tick);
42
43#elif AETM_TICK_ROUNDING_METHOD == 1 // just round
44 #define _AETM_TICK_ROUND(tick) ace::math::roundToInt<ullint>(tick);
45
46#elif AETM_TICK_ROUNDING_METHOD == 2 // round up/ceil
47 #define _AETM_TICK_ROUND(tick) ace::math::ceilToInt<ullint>(tick);
48#else

50#endif

52/// <summary>
53/// This module is the engine's **tick timer**.
54///
55/// It allows to time or predict events ("happen at tick X"), with the variable tick speed. <br>
56/// Internally it uses the AEFrame module for the delay, so the same limitations apply.
57///
58/// Internally it makes a separate thread that increments the tick number every set time period, and sleeping after.
59///
60/// @see AEFrame
61/// @note This is multithreading-safe! :D
62/// </summary>
63class AETimer : public __AEModuleBase<AETimer> {
64
65public:
66

68 /// <summary>
69 /// Class constructor -- **sets the delay** in ticks per second, and **starts the thread**.
70 /// </summary>
71 /// <param name="tps">The amount of ticks per second, as the double</param>
72 explicit AETimer(const double tps) :
73 m_rDelay(tps), m_ullTicks(0), m_bRunTrd(false) {
74 this->startThread();
75 }
76
77 /// <summary>
78 /// Class constructor -- assigns **default delay (ENGINE_FPS) and doesn't start the thread**.
79 /// </summary>
80 AETimer(void) noexcept :
81 m_rDelay(ENGINE_FPS), m_ullTicks(0), m_bRunTrd(false)
82 {}

84 /// <summary>
85 /// Copy constructor -- **copies the data** and, if the original timer was started, **starts the timer**.
86 /// </summary>
87 /// <param name="tm">The original AETimer instance to make a copy of</param>
88 AETimer(const AETimer& tm) :
89 //COPYCONSTRUCT_MODULE(AETimer, tm),
90 m_rDelay(tm.m_rDelay), m_ullTicks(tm.m_ullTicks.load()), m_bRunTrd(false) {
91
92 if (tm.m_bRunTrd) {
93 this->startThread();
94 }
95 }

97 /// <summary>
98 /// Copy assignment operator -- **copies the data** and, if the original timer was started, **starts the timer**.
99 /// </summary>
100 /// <param name="tm">The original AETimer instance to make a copy of</param>
101 /// <returns>Reference to the resulting AETimer copy</returns>
102 AETimer& operator=(const AETimer& tm) {
103 this->stopThread();
104 this->m_rDelay = tm.m_rDelay;
105 this->m_ullTicks.store(tm.m_ullTicks.load());
106 if (tm.m_bRunTrd) {
107 this->startThread();
108 }
109 return *this;
110 }

112 /// <summary>
113 /// **Class destructor**
114 /// </summary>
115 ~AETimer(void) {
116 this->stopThread();
117 }

119 /// <summary>
120 /// **Starts the timer thread** and starts tick counting.
121 ///
122 /// @note Does nothing if thread is already started
123 /// @note Does nothing if the previously-set delay/fps is invalid.
124 /// @see AEFrame::isValidFPS()
125 /// @see AEFrame::isValidDelay()
126 /// @see AEFrame::hasDelay()
127 /// </summary>
128 inline void startThread(void) {
129 if (!this->m_rDelay.hasDelay()) {
130 return; // invalid fps
131 }
132
133 if (this->m_bRunTrd.exchange(true)) {
134 return; // we're already running, silly
135 }
136
137 this->m_trdCounting = std::thread(&AETimer::tickTimer, this);
138 if (!this->m_trdCounting.joinable()) {
139 throw std::runtime_error("Could not start AETimer thread!");
140 }
141 }
142
143 /// <summary>
144 /// **Stops the timer thread** and stops tick counting.
145 ///
146 /// @note Does nothing if the thread was already stopped.
147 /// </summary>
148 inline void stopThread(void) {
149 if (!this->m_bRunTrd.exchange(false)) {
150 return; //maybe it was closed already?
151 }
152
153 this->m_trdCounting.join();
154 }
155
156 /// <summary>
157 /// Sets the timer's **new tick value to the passed value**.
158 /// </summary>
159 /// <param name="tick">The tick value to set instance's tick value to</param>
160 inline void setTick(const ullint tick) noexcept {
161 this->m_ullTicks.store(tick, std::memory_order::relaxed);
162 }
163
164 /// <summary>
165 /// Sets the **timer tick back to 0**.
166 /// </summary>
167 inline void resetTick(void) noexcept {
168 this->setTick(0);
169 }
170
171
172 /// <summary>
173 /// Sets the timer's **new delay as ticks-per-second**.
174 ///
175 /// @attention If the value invalid, stops the thread.
176 /// @see AEFrame::isValidFPS()
177 /// @note Uses a mutex, so there *could* be some delay. Check the mutex behaviour on your system.
178 /// </summary>
179 /// <param name="tps">The amount of ticks per second to set the delay to</param>
180 /// <param name="flags">
181 /// The flag to indicate actions after setting the fps:
182 /// * **0** -- does nothing
183 /// * **1** -- reset the tick amount after changing the ticks-per-second value
184 /// * **2** -- approximately scale the tick counter to try to keep the world-time consistent; Uses the method of rounding indicated by #AETM_TICK_ROUNDING_METHOD
185 ///
186 /// @attention If the previously-set delay/fps was invalid, it resets the tick value to 0, regardless of the passed **flags** value.
187 /// </param>
188 inline void setTPS(const double tps, const cint flags = 0) {
189 std::lock_guard mtx(this->m_mtxThreadMutex);
190
191 if (!AEFrame::isValidFPS(tps)) {
192 this->stopThread();
193 }
194 else {
195 if (this->m_rDelay.hasDelay() || flags == 1) {
196 if (flags == 2) {
197 this->m_ullTicks = _AETM_TICK_ROUND((this->m_ullTicks * (tps / this->getTPS())));
198 }
199 }
200 else {
201 this->resetTick();
202 }
203
204 }
205 this->m_rDelay.setFps(tps);
206 }
207
208 /// <summary>
209 /// Checks **whether the timer is ticking** and the thread is running.
210 /// </summary>
211 /// <returns>
212 /// If the thread was started/is running:
213 /// * **True**
214 ///
215 /// Otherwise:
216 /// * **False**
217 /// </returns>
218 inline bool isTicking(void) const noexcept {
219 // Note to future engine maintainers: do NOT put this in place of the
220 // m_bRunTrd.exchange(), unless you want implicit race conditions
221 // This is meant only for the user.
222 return this->m_bRunTrd.load(std::memory_order::relaxed);
223 }
224
225 /// <summary>
226 /// Returns the **current tick** of the timer.
227 ///
228 /// @note If thread is not started/working, the return value will be the same (duh)
229 /// </summary>
230 /// <returns>
231 /// Current timer tick as **ullint** type.
232 /// </returns>
233 inline ullint getTick(void) const noexcept {
234 return this->m_ullTicks.load(std::memory_order::relaxed);
235 }
236
237 /// <summary>
238 /// Calculates the *approximate world time* of the timer's ticks as seconds.
239 ///
240 /// @note If thread is not started/working, the return value will be the same (duh)
241 /// </summary>
242 /// <returns>
243 /// The approximation of the counted world time in seconds using the timer's ticks, as the **double** type.
244 /// </returns>
245 inline long double getWorldTime(void) const noexcept {
246 return this->m_ullTicks.load(std::memory_order::relaxed) * (long double) this->m_rDelay.getDelay();
247 }
248
249 /// <summary>
250 /// Returns the **ticks-per-second** of the AETimer's instance (aka AEFrame's fps goal).
251 ///
252 /// @see AEFrame::getFPS()
253 /// </summary>
254 /// <returns>
255 /// The ticks-per-second rate of the instance as **double** type.
256 /// </returns>
257 inline double getTPS(void) const noexcept {
258 return this->m_rDelay.getFPS();
259 }
260
261 /// <summary>
262 /// Returns the **delay-per-tick** of the instance, as seconds (aka AEFrame's delay).
263 ///
264 /// @see AEFrame::getDelay()
265 /// </summary>
266 /// <returns>
267 /// Delay-per-tick of the instance as seconds, as the **double** type.
268 /// </returns>
269 inline double getDelay(void) const noexcept {
270 return this->m_rDelay.getDelay();
271 }
272
273 /// <summary>
274 /// Stops and **sleeps the thread untill** the given timer hits **the certain tick number**.
275 ///
276 /// </summary>
277 /// <param name="timer">The instance of the AETimer to wait for</param>
278 /// <param name="tick">The AETimer instance tick value to wait for</param>
279 static inline void waitForTick(const AETimer& timer, const ullint tick) noexcept {
280 AEFrame fr(timer.getTPS()*2);
281 while (tick > timer.getTick()){
282 fr.sleep();
283 }
284 }
285
286 /// <summary>
287 /// Stops and **sleeps the thread untill** the given timer **goes through certain amount of ticks**.
288 ///
289 /// </summary>
290 /// <param name="timer">The instance of the AETimer to wait for</param>
291 /// <param name="tick">The amount of ticks to wait for in the AETimer instance</param>
292 static inline void waitTicks(const AETimer& timer, const ullint tick) noexcept {
293 AETimer::waitForTick(timer, tick + timer.m_ullTicks);
294 }

296
297private:
298
299 /// <summary>
300 /// The function that **does the tick counting** for the timer.
301 /// </summary>
302 inline void tickTimer(void) noexcept {
303 //load the atomic in the loop as fast as possible
304 //we don't care about a lot of sync, since...it's 1 byte
305 //and we just need approximate time to stop the timer
306 while (this->m_bRunTrd.load(std::memory_order_relaxed)) {
307 this->m_rDelay.sleep();
308 this->m_ullTicks.fetch_add(1, std::memory_order::relaxed);
309 }
310 }
311
312 /// The **frame-rater of the timer** for proper delay in the tick-counting function loop.
313 /// @see AETimer::tickTimer()
314 AEFrame m_rDelay;
315 /// The thread object to **launch the thread** for the tick counting.
316 std::thread m_trdCounting;
317 /// The mutex to **syncronise changes with the AETimer::m_rDelay** when we're touching that.
318 std::mutex m_mtxThreadMutex;
319 /// The **tick counter** variable.
320 std::atomic<ullint> m_ullTicks;
321 /// The **flag to continue the loop** for tick counting.
322 std::atomic<bool> m_bRunTrd;
323
324 //aaaand yeah, register the class
325 /// <summary>
326 /// The instantiation of necessary functions from __AEModuleBase class.
327 ///
328 /// This is things like module names, etc.
329 /// @warning This is **NOT** a function of this class. Instead it's a macro call to the #REGISTER_MODULE() in the AEModuleBase.hpp.
330 /// </summary>
331 REGISTER_MODULE(AETimer);
332};
