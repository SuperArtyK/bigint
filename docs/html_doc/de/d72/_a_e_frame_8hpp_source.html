37/// <summary>
38/// This module is the engine's **precise frame/loop-cycle limiter**.
39///
40/// It allows to limit the execution of a loop to a set amount of iterations per second. <br>
41/// Or, instead have the set delay per each loop cycle.
42///
43/// It works by sleeping the difference between the time spent by code in the loop and the time per each frame.
44///
45/// For example: <br>
46/// If you want the loop to have 100 cycles/second or fps (10 milliseconds per cycle),
47/// and the code takes **4ms** to complete, then the AEFrame will sleep the difference. <br>
48/// In this case it's **6ms** to make it up to the 10ms per cycle.
49///
50/// @note By default it uses the value of #ENGINE_FPS as the fps target for the loops
51/// @note This isn't multithreading safe.
52/// </summary>
53class AEFrame : public __AEModuleBase<AEFrame> {
54
55public:

57 /// This is an an **alias for the internal clock** of the AEFrame module.
58 /// @warning The resulution of the AEFrameClock that your compiler gives
59 /// must be at least 1 microsecond, otherwise it won't compile (breaks assertion).

63 /// <summary>
64 /// Class constructor -- constructs the instance with the **fps as the delay**.
65 ///
66 /// @see AEFrame::setFPS()
67 /// @attention If you pass the invalid fps value, it will disable the delay
68 /// (sets AEFrame::m_bNoDelay to **true**); Setting it to a valid value enables it back.
69 /// @see AEFrame::isValidFPS()
70 /// @note Defaults the fps delay to #ENGINE_FPS
71 /// </summary>
72 /// <param name="fps">The fps goal that the frame delay module will try to achieve</param>
73 explicit AEFrame(const double fps = ENGINE_FPS) noexcept :
74 m_tdChronoDelay((!AEFrame::isValidFPS(fps)) ? microSec(0) : microSec(llint(1000000.0 / fps))),
75 m_tpTimepoint(AEFrameClock::now()), m_fFPS(fps), m_bNoDelay((fps <= 0 || fps > 1000000.0))
76 {}

109 /// <summary>
110 /// Sets the delay of the instance as the **frames (loop cycles) per second** to reach in execution (fps goal).
111 ///
112 /// @attention If you pass the invalid fps value, it will disable the delay
113 /// (sets AEFrame::m_bNoDelay to **true**); Setting it to a valid value enables it back.

142 /// <summary>
143 /// Puts the current thread to **sleep for the delay that was set earlier**.
144 ///
145 /// Example: if the fps delay was set to 30fps (~~33ms delay in real time) and a game
146 /// loop takes just 13ms, then the thread will sleep the rest of 20ms to make the
147 /// game loop run at around ~~33ms of delay or around 30fps.
148 ///
149 /// @attention If the delay is disabled (AEFrame::m_bNoDelay is **true**), just returns and doesn't do anything
150 /// @note If the AEFrame::m_tpTimepoint gets too out of touch with real time, resets the time point (calls AEFrame::resetTimePoint())
151 /// The definition of "out of touch" is controlled by #AEF_RESET_TIME_WHEN_BEHIND_MS.
152 /// @note Automatic timepoint resetting works **only** if #AEF_RESET_TIME_WHEN_BEHIND_MS is **enabled**
153 /// </summary>
154 inline void sleep(void) noexcept {
155 // add to time point
156 if (this->m_bNoDelay) {
157 return;
158 }
159#if AEF_RESET_TIME_WHEN_BEHIND_MS > 0
160 if ((AEFrameClock::now() - this->m_tpTimepoint) > m_tdResetTime) {
161 this->resetTimePoint();
162 }
163#endif
164
165 this->m_tpTimepoint += this->m_tdChronoDelay;
166 // and sleep until that time point
167 sleepUntil(this->m_tpTimepoint);
168 }
169
170 /// <summary>
171 /// **Resets the time point** in the AEFrame to the real (current) time.
172 ///
173 /// Helps if the AEFrame::m_tpTimepoint time point drifted far from the reality.
174 /// </summary>
175 inline void resetTimePoint(void) noexcept {
176 this->m_tpTimepoint = AEFrameClock::now();
177 }

201 /// <summary>
202 /// Returns whether the instance **has a delay or not** (invalid fps/delay was set).
203 ///
204 /// @see AEFrame::isValidDelay()
205 /// </summary>
206 /// <returns>
207 /// If the instance has delay enabled (valid fps/delay):
208 /// * **True**
209 ///
210 /// If the instance has delay disabled (invalid fps/delay):
211 /// * **False**
212 /// </returns>
213 inline bool hasDelay(void) const noexcept {

217 /// <summary>
218 /// Returns whether the passed **delay as fps is valid**.
219 /// </summary>
220 /// <param name="fps">The frames per second value to check</param>
221 /// <returns>
222 /// If the fps value is valid -- more than 0 and less than/equal to 1 million:
223 /// * **True**
224 ///
225 /// Otherwise:
226 /// * **False**
227 /// </returns>
228 static constexpr bool isValidFPS(const double fps) noexcept {
229 return (fps > 0.0 && fps <= 1000000.0);
230 }
231
232 /// <summary>
233 /// Returns whether the passed **delay as seconds is valid**.
234 /// </summary>
235 /// <param name="fps">The frames per second value to check</param>
236 /// <returns>
237 /// If the delay value is valid -- more than 0 and less than int32_t max (2^31 - 1):
238 /// * **True**
239 ///
240 /// Otherwise:
241 /// * **False**
242 /// </returns>
243 static constexpr bool isValidDelay(const double sec) noexcept {
244 return (sec > 0.0 && sec <= (double)std::numeric_limits<int32_t>::max());
245 }
